    \begin{frame}{How:}
        \centering
        Shuffling Pixel Values, mod 256:
        \begin{equation*}
            f ( p ) := p_{i+1} = \lambda_{1} * p_{i} ( 1 - p_{i} )
        \end{equation*}
        Shuffling pixel locations, mod (image size)
        \begin{equation*}
            g ( p ) := p_{i+1} = \lambda_{2} * p_{i} ( 1 - p_{i} )
        \end{equation*}
    \end{frame}
